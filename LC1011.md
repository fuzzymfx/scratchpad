## Brute-force solution: iteratve over the range: max weight in the array to summ of all the weights

```js
class Solution {
public:
    int shipWithinDays(vector<int>& weights, int days) {
        int mincap = *max_element(weights.begin(), weights.end());       
        int maxcap = accumulate(weights.begin(),weights.end(), 0);
        int ans =0;

        for(int i =mincap;i<=maxcap; i++ ){
            int sum =0;
            int j =0;
            int count =0;
            // cout<<endl<<endl;
            while(j<weights.size()){
                
                sum+=weights[j];
                if(j==weights.size()-1){
                    if(sum<=i){
                        count++;
                        break;
                    }
                }
                if(sum==i){
                    count++;
                    sum=0;
                    // cout<<weights[j]<<endl;
                }
                else if(sum>i){
                    count++;
                    sum=0;
                    j--;
                    // cout<<endl;
                }
                // else cout<<weights[j]<<" ";

                j++;
            }
            // cout<<count<<" "<<i<<endl;

            if(count<=days)
            return i;

        }
        return ans;
    }
};
```

## Binary Search: Instead of iterating over the entire range, realize that once you're finding out a solution, the other solutions greater than the sol are worthless. So, you can use binary search to find the solution

```js
class Solution {
public:
    int shipWithinDays(vector<int>& weights, int days) {
        int mincap = *max_element(weights.begin(), weights.end());       
        int maxcap = accumulate(weights.begin(),weights.end(), 0);
        int ans =maxcap;
        cout<<mincap<<" "<<maxcap;

        while(mincap<maxcap){
            int mid = (mincap+maxcap)/2;
            // cout<<mincap<<" "<<mid<<" "<<maxcap<<" "<<ans<<endl;
            int sum =0;
            int j =0;
            int count =0;
            // cout<<endl<<endl;
            while(j<weights.size()){
                
                sum+=weights[j];
                if(j==weights.size()-1){
                    if(sum<=mid){
                        count++;
                        break;
                    }
                }
                if(sum==mid){
                    count++;
                    sum=0;
                    // cout<<weights[j]<<endl;
                }
                else if(sum>mid){
                    count++;
                    sum=0;
                    j--;
                    // cout<<endl;
                }
                // else cout<<weights[j]<<" ";

                j++;
            }
            // cout<<count<<" "<<i<<endl;

            if(count<=days){
                maxcap=mid;
                ans=std::min(ans, mid);
            }
            else{
                mincap=mid+1;
            }
        }
        return ans;
    }
};
```
